def model_lr(X, Y, learning_rate, num_iterations, random_w_b=False):
    if random_w_b:
        W, b = np.random.randn(1, X.shape[0]) * 0.01, np.random.randn() * 0.01
    else:
        W, b = np.zeros((1, X.shape[0])), 0
    m = X.shape[1]
    for i in tqdm(range(num_iterations)):
        A = sigmoid(W @ X + b)
        cost = -np.sum(Y * np.log(A) + (1 - Y) * np.log(1 - A)) / m
        dW = (A - Y) @ X.T / m
        db = np.sum(A - Y) / m
        W -= learning_rate * dW
        b -= learning_rate * db
    return W, b

def model_nn(X, Y, learning_rate, num_iterations, n_h, random_w_b=False):
    if random_w_b:
        W1, b1 = np.random.randn(n_h, X.shape[0]) * 0.01, 0
        W2, b2 = np.random.randn(1, n_h) * 0.01, 0
    else:
        W1, b1 = np.zeros((n_h, X.shape[0])), 0
        W2, b2 = np.zeros((1, n_h)), 0
    m = X.shape[1]
    for i in tqdm(range(num_iterations)):
        A1 = np.tanh(W1 @ X + b1)
        A2 = sigmoid(W2 @ A1 + b2)
        cost = -np.sum(Y * np.log(A2) + (1 - Y) * np.log(1 - A2)) / m
        dZ2 = A2 - Y
        dW2 = dZ2 @ A1.T / m
        db2 = np.sum(dZ2, axis=1, keepdims=True) / m
        dZ1 = W2.T @ dZ2 * (1 - np.power(A1, 2))
        dW1 = dZ1 @ X.T / m
        db1 = np.sum(dZ1, axis=1, keepdims=True) / m
        W1 -= learning_rate * dW1
        b1 -= learning_rate * db1   
        W2 -= learning_rate * dW2
        b2 -= learning_rate * db2
    return W1, b1, W2, b2
